class Solution {
    public int reverse(int x) {
        long rev=0;int rem=0;
        while(x!=0){
            rem=x%10;
            rev=rev*10+rem;
            x/=10;
        }
        if(rev<2147483647 && rev>-2147483648)
            rem=(int)rev;
        else
            rem=0;
        return rem;
    }
}
